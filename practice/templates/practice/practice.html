const addButton = document.querySelector('#section-18 button')

/*

    itemContainer.appendChild(newItem)
})

Event 문제 2
링크입니다 
const linkInput = document.querySelector('#section-19 input') 
const link = document.querySelector('#section-19 a') 

/*
2. 링크를 클릭했을때, input태그에 입력된 주소로 이동하도록 해주세요. 
- HINT : input태그에 입력된 value를 a태그 href 속성에 넣어줍니다.
*/


linkInput.addEventListener('input', (e) => { 
    link.href = e.target.value 
})
